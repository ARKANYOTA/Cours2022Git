def localise_max(L):
    c = 0
    i = []
    for k in range(len(L)):
        if L[k] > c:
            i.clear()
            i.append(k)
            c = L[k]
        elif L[k] == c:
            i.append(k)
    return i

def transpose(T):
    nv_T = [[] for i in range(len(T[1]))]
    for ii, i in enumerate(T):
        for ij, j in enumerate(i):
            nv_T[ij] += [j]
    return nv_T

def nb_elt(T) -> int:
    count = 0
    for i in T:
        if isinstance(i, list):
            count += nb_elt(i)
        else:
            count += 1
    return count
